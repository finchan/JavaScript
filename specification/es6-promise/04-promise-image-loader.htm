function loadImageAsync(url) { return new Promise((resolve, reject) => { var image = new Image(); image.onload = function() { resolve(image) } image.onerror = function() { reject(new Error('Could not load image at ' + url)) } image.src = url }) } loadImageAsync('https://www.baidu.com/img/flexible/logo/pc/result.png').then((value)=>{ console.log(["Success", value]) }, (error) =>{ console.log(["Failed", error]) })
